def get_top3(subf):
    top3 = subf.sort_values(by='revenue', ascending=False)[:3]
    reshaped_revenue = top3['revenue'].values.reshape(1,3)
    revenues = pd.Series(data = reshaped_revenue[0], index = ['top1','top2','top3'])
    dict_top3[subf.name] = revenues
